# Tabla descriptiva general
tabla_general <- datos %>% 
  summarise(
    Grupo = "Total",
    n = length(Edad),
    Promedio = mean(Edad, na.rm = TRUE),
    DS = sd(Edad, na.rm = TRUE),
    Mediana = median(Edad, na.rm = TRUE),
    RIC = IQR(Edad, na.rm = TRUE),
    Minimo = min(Edad, na.rm = TRUE),
    Maximo = max(Edad, na.rm = TRUE)
  )

# Tabla descriptiva por grupo
tabla_grupos <- datos %>% 
  group_by(Grupo) %>% 
  summarise(
    n = length(Edad),
    Promedio = mean(Edad, na.rm = TRUE),
    DS = sd(Edad, na.rm = TRUE),
    Mediana = median(Edad, na.rm = TRUE),
    RIC = IQR(Edad, na.rm = TRUE),
    Minimo = min(Edad, na.rm = TRUE),
    Maximo = max(Edad, na.rm = TRUE),
    .groups = "drop"
  )

# Unir tablas
tabla_final <- bind_rows(tabla_general, tabla_grupos)

datos %>% 
  ggplot(aes(x = Grupo, y = Edad)) +
  geom_violin(fill = "#A8DADC", alpha = 0.6, color = NA) +  # tono pastel
  geom_boxplot(width = 0.15, fill = "#A8DADC", alpha = 0.9,
               outlier.color = "gray40") +
  stat_summary(fun = mean, geom = "point",
               shape = 21, size = 3, fill = "white", color = "black") +
  labs(
    title = "Distribución de la Edad por Grupo",
    x = "Grupo",
    y = "Edad"
  ) +
  theme_minimal() +
  theme(
    legend.position = "none",
    plot.title = element_text(face = "bold", size = 14),
    axis.text = element_text(size = 10)
  )
